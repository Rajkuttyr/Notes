class Solution {
    List<List<Integer>> res = new ArrayList<>();
    int[] nums;

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        Arrays.sort(candidates); // VERY IMPORTANT
        nums = candidates;
        Backtrack(0, 0, new ArrayList<>(), target);
        return res;
    }

    void Backtrack(int ind, int sum, List<Integer> temp, int target) {

        if(sum == target){
            res.add(new ArrayList<>(temp));
            return;
        }

        if(ind == nums.length || sum > target) return;

        // include current element
        temp.add(nums[ind]);
        Backtrack(ind + 1, sum + nums[ind], temp, target);
        temp.remove(temp.size() - 1);

        // skip duplicates
        while(ind + 1 < nums.length && nums[ind] == nums[ind + 1]){
            ind++;
        }

        // exclude
        Backtrack(ind + 1, sum, temp, target);
    }
}
