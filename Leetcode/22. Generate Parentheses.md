class Solution {

List<String> list=new ArrayList<>();

public List<String> generateParenthesis(int n) {

rec("",n,0,0);

return list;

  

}

public void rec(String str,int n,int op,int cp ){

if(op==n&&cp==n){

list.add(str);

}

  

if(op<n){

rec(str+"(",n,op+1,cp);

  

}

if(op>cp){

rec(str+")",n,op,cp+1);

}

}

}
