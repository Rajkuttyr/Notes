class Solution {

List<List<Integer>> res = new ArrayList<>();

int[] nums;

public List<List<Integer>> combinationSum(int[] candidates, int target) {

nums=candidates;

List<Integer> temp = new ArrayList<>();

Backtrack(0,0,temp,target);

return res;

}

void Backtrack(int ind,int s,List<Integer> temp,int sum){

if(s==sum){

res.add(new ArrayList<>(temp));

return ;

}

if(ind==nums.length ||s>sum) return ;

temp.add(nums[ind]);

s+=nums[ind];

Backtrack(ind,s,temp,sum);

temp.remove(temp.size()-1);

s-=nums[ind];

Backtrack(ind+1,s,temp,sum);

}

}
