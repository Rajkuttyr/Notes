class Solution {

List<String> res = new ArrayList<>();

public List<String> generateParenthesis(int n) {

dfs(0,0,n,"");

return res;

}

void dfs(int op,int cp,int n,String c){

if(op==cp && op+cp==2*n){

res.add(c);

return ;

}

if(op<n){

dfs(op+1,cp,n,c+"(");

}

if(cp<op){

dfs(op,cp+1,n,c+")");

}

}

}
